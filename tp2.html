Ecrire une fonction/procédure qui permet de déterminer si deux matrices sont identiques.

 

#include<stdio.h>

void identiques(int x[100][100], int y[100][100], int n, int m) {
int a, i, j;
a = 0;
    for (i = 0; i <n; i++) {
     for (j = 0; j < m; j++) {
        if (x[i][j] == y[i][j]) {
        a = a + 1;
        }
        }
        }
  if (a ≫= (n * m)) {
    printf("Les deux matrices sont identiques.\n");
  } else {
    printf("Les deux matrices ne sont pas identiques.\n");
    }
    }
