Ecrire un programme qui permet de déterminer le nombre de succession de deux valeurs particulier dans T1.

 

#include <stdio.h>

 int main() {
    int n=100;
    int t[n];
    int v2,v1,x,i;
    printf("entrer n: ");
    scanf("%d",&n);
    x=0;
    for(i=0;i<n;i++){
        printf("t[%d]: ",i);
        scanf("%d",&t[i]);
    }
    printf("entrer v1: ");
    scanf("%d",&v1);
    printf("entrer v2: ");
    scanf("%d",&v2);
    for (i=0;i<n-1;i++){
        if (v1==t[i] && v2==t[i+1]){
            x=x+1;
        }
        if (v1==t[i+1] && v2==t[i]){
            x=x+1;
        }
    }
    if (x>0){
        printf("NB succession de v1=%d et v2=%d est %d\n",v1,v2,x);
    } else {
        printf("v1=%d et v2=%d ne sont pas successifs \n",v1,v2);
    }
    return 0;
}

Ecrire un programme qui permet de calculer la moyenne des valeurs situé dans le diagonal de T2.

 
    
#include<stdio.h>

 int main()
 {
    int x=100,y=100,n;
    int t2[x][y];
    int j,i;
    float s,M;

    printf("Entrez la taille de t2: ");
    scanf("%d",&n);

    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            printf("Entrez la valeur de t2[%d][%d]: ",i,j);
            scanf("%d",&t2[i][j]);
        }
    }

    s=0;
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            if (i==j){
                s=s+t2[i][j];
            }
        }
    }
    M =s/n;
    printf("La moyenne  diagonale de T2 est %f", M);
    return 0;
}
