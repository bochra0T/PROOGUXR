                    I = 0;
                    break;
                }

5.2.Ecrire un programme qui lit deux mots et qui détermine s’il sont anagramme.


#include <stdio.h>
    #include <string.h>
    
    void ch(char m[]) {
        int i, j, n = strlen(m);
        char a;
        for(i = 0; i < n-1; i++) {
            for(j = i+1; j < n; j++) {
                if(m[i] > m[j]) {
                    a = m[i];
                    m[i] = m[j];
                    m[j] = a;
                }
            }
        }
    }
    
    int anagramme(char m1[], char m2[]) {
        int i, n = strlen(m1);
        if(n != strlen(m2)) {
            return 0;
        }
        ch(m1);
        ch(m2);
        for(i = 0; i < n; i++) {
            if(m1[i] != m2[i]) {
                return 0;
            }
        }
        return 1;
    }
